11 private const val INT_INTERMEDIATE = 1000

13 private val LINE_SEPARATOR = System.getProperty("line.separator")

15 internal object RisExport {
16 
17     /**
18      * Converts a flow of [RisRecord]s into a flow of [String]s in RIS file format.
19      * Optionally accepts a list of names of [RisTag]s defining a sort order for the [RisTag]s in the file.
20      */
21     internal fun build(recordFlow: Flow<RisRecord>, sort: List<String> = emptyList()): Flow<String> = flow {
22         fun RisTag.withValue(value: Any): String = "$this$TAG_SEPARATOR$value$LINE_SEPARATOR"
23         val sortMap = sort.withIndex().associate { RisTag.valueOf(it.value) to it.index }.toMap()
24         var firstRecord = true
25         recordFlow.collect { risRecord ->
26             if (!firstRecord) emit(LINE_SEPARATOR)
27             firstRecord = false
28             RisTag.values().sortedWith(sortMap.toComparator()).forEach { tag ->
29                 tag.getFrom(risRecord)?.let { recordValue: Any ->
30                     when (recordValue) {
31                         is List<*> -> recordValue.forEach { listValue ->
32                             emit(tag.withValue(listValue as String))
33                         }
34                         is String -> emit(tag.withValue(recordValue.truncatedTo(tag.maxLength)))
35                         else -> emit(tag.withValue(recordValue))
36                     }
37                 }
38             }
39         }
40     }
41 
42     /**
43      * Sorting
44      *
45      * * first by fixOrder (where specified), guaranteeing e.g. TY is first and ER is last
46      * * then by explicit order as provided to the build call
47      * * lastly by tag name
48      */
49     private fun Map<RisTag, Int>.toComparator(): Comparator<RisTag> =
50         compareBy({ it.requiredOrder }, { this[it] ?: INT_INTERMEDIATE }, { it.name })
51
