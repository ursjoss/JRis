12 import kotlinx.coroutines.flow.filterNotNull
13 import kotlinx.coroutines.flow.flow
14 
15 private val TAG_LENGTH = RisTag.ER.name.length
16 private val START_INDEX_VALUE = TAG_LENGTH + TAG_SEPARATOR.length
17 
18 internal object RisImport {
19 
20     /**
21      * Converts a flow of [String]s, conforming to the RIS format into a flow of [RisRecord]s.
22      */
23     internal fun process(lineFlow: Flow<String>): Flow<RisRecord> = flow {
24         var record = RisRecord()
25         var previousTag: RisTag? = null
26 
27         lineFlow.filterNotNull()
28             .filterNot { line -> line.isEmpty() }
29             .collect { line ->
30                 if (line.isEndOfRecord()) {
31                     emit(record)
32                     record = RisRecord()
33                 } else {
34                     previousTag = record.fillFrom(line, previousTag)
35                 }
36             }
37     }
38 
39 
40     private fun String.isEndOfRecord() = startsWith(RisTag.ER.name)
41 
42     /**
43      * Fills the tag and the respective value into the [RisRecord] (receiver).
44      * Returns the tag as context for parsing the next line.
45      */
46     private fun RisRecord.fillFrom(line: String, previousTag: RisTag?): RisTag? {
47         if (line.length <= START_INDEX_VALUE + 1)
48             return previousTag
49 
50         val tagName = line.substring(0, TAG_LENGTH)
51         val tag = RisTag.fromName(tagName)
52             ?: if (previousTag == RisTag.AB) RisTag.AB
53             else throw KRisException("Unable to parse tag '$tagName'")
54         tag.setInto(this, tag.typeSafeValueFrom(line))
55         return tag
56     }
57 
58     private fun RisTag.typeSafeValueFrom(line: String): Any? {
59         val rawValue: String = line.substring(START_INDEX_VALUE).trim()
60         return when (kClass) {
61             RisType::class -> RisType.valueOf(rawValue)
62             Long::class -> rawValue.toLong()
63             else -> rawValue.truncatedTo(maxLength)
64         }
65     }
66 }
