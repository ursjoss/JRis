21 /**
22  * `RIS` format processor and builder. It is capable of
23  *
24  * * processing lines of RIS files (Strings), converting them to [RisRecord]s
25  * * building well formatted RIS files from [RisRecord]s.
26  *
27  * The [KRis] class works in non-blocking manner as default.
28  * Extension functions provide blocking alternatives for ease of use both from Kotlin or Java.
29  *
30  * @author Gianluca Colaianni -- g.colaianni5@gmail.com
31  * @author Urs Joss - urs.joss@gmx.ch
32  */
33 @Suppress("KDocUnresolvedReference")
34 @ExperimentalCoroutinesApi
35 object KRis {
36 
37 //region:process - RISFile lines -> RisRecords
38 
39     /**
40      * Converts a flow of Strings (representing lines in a RIS file) into a flow of [RisRecord]s.
41      * May throw a [KRisException] if the line flow cannot be parsed successfully.
42      */
43     fun process(lineFlow: Flow<String>): Flow<RisRecord> = RisImport.process(lineFlow)
44 
45     /**
46      * Converts an observable of Strings (representing lines in a RIS file) into an observable of [RisRecord]s
47      * in non-blocking manner. May throw [KRisException] if the line flow cannot be parsed successfully.
48      */
49     @JvmStatic
50     fun processObservables(risLineObservable: Observable<String>): Observable<RisRecord> =
51         process(risLineObservable.asFlow()).asObservable()
52 
53     /**
54      * Converts a list of Strings (representing lines in a RIS file) into a list of [RisRecord]s in blocking manner.
55      * May throw a [KRisException] if the line flow cannot be parsed successfully.
56      */
57     @JvmStatic
58     fun processList(risLines: List<String>): List<RisRecord> = runBlocking { process(risLines.asFlow()).toList() }
59 
60 //endregion
61 
62 //region:build - or RisRecords -> RISFile lines
63 
64     /**
65      * Converts a flow of [RisRecord]s into a flow of [String]s in RIS file format.
66      * Optionally accepts a list of names of [RisTag]s defining a sort order for the [RisTag]s in the file.
67      */
68     fun build(recordFlow: Flow<RisRecord>, sort: List<String> = emptyList()): Flow<String> = RisExport.build(recordFlow, sort)
69 
70     /**
71      * Converts a list of [RisRecord]s into a list of [String]s in RIS file format in blocking manner.
72      * Optionally accepts a list of names of [RisTag]s defining a sort order for the [RisTag]s in the file.
73      */
74     @JvmStatic
75     @JvmOverloads
76     fun buildFromList(risRecords: List<RisRecord>, sort: List<String> = emptyList()): List<String> =
77         runBlocking { build(risRecords.asFlow(), sort).toList() }
78 
79     /**
80      * Converts an observable of [RisRecord]s into an observable of [String]s in RIS file format.
81      * Optionally accepts a list of names of [RisTag]s defining a sort order for the [RisTag]s in the file.
82      */
83     @JvmStatic
84     @JvmOverloads
85     fun exportObservable(observable: Observable<RisRecord>, sort: List<String> = emptyList()): Observable<String> =
86         build(observable.asFlow(), sort).asObservable()
87 
88 //endregion
89 }
